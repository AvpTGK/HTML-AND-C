Different Patterns (square, rectangle, right pyramid, inverse right pyramid)


Code:


#include 

int main() {
    int i, j;
    char a[5][4];

    printf("Square and Rectangle \n");
    for (i = 0; i < 5; i++) {
        for (j = 0; j < 4; j++) {
            a[i][j] = 'A';
        }
    }

    for (i = 0; i < 5; i++) {
        for (j = 0; j < 4; j++) {
            printf("%c", a[i][j]);
        }
        printf("\n");
    }
    printf("\n");

    int k, l;
    char b[5][5];

    for (k = 0; k < 5; k++) {
        for (l = 0; l < 5; l++) {
            b[k][l] = 'A';
        }
    }

    for (k = 0; k < 5; k++) {
        for (l = 0; l < 5; l++) {
            printf("%c", b[k][l]);
        }
        printf("\n");
    }

    int rows = 5;

    // Right Pyramid
    printf("\nRight Pyramid:\n");
    for (int t = 1; t <= rows; t++) {
        for (int v = 1; v <= rows - t; v++) {
            printf(" ");
        }
        for (int v = 1; v <= t; v++) {
            printf("* ");
        }
        printf("\n");
    }

    // Right Inverse Pyramid
    printf("\nRight Inverse Pyramid:\n");
    for (int t = rows; t >= 1; t--) {
        for (int v = 1; v <= rows - t; v++) {
            printf(" ");
        }
        for (int v = 1; v <= t; v++) {
            printf("* ");
        }
        printf("\n");
    }

    return 0;
}
